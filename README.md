{ Assuming you have Python 3.x installed }

vituralenv setup:
	> python3 -m venv /.src

activate vituralenv:
	> source .src/bin/activate

install requirements:
	> pip install -r requirements.txt

launch django application:
	> python manage.py runserver 0.0.0.0:8000

visit app in a browser:
    > http://0.0.0.0:8000/
